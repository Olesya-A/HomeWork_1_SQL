CREATE DATABASE phones;

USE phones;

CREATE TABLE MobilePhone
(
Id_phone INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
ProductName VARCHAR(30) NOT NULL,
Manufacturer VARCHAR(30) NOT NULL,
ProductCount VARCHAR(30) NOT NULL,
Price VARCHAR(30)
);
